car and cdr

  • car is the first element in the list
  • cdr is the rest
(cdr '(1 2 3))

CDR-ing down a list

(defun apply-fun(fun l)
   (if (null l)
      nil
      (cons (funcall fun (car l))
            (apply-fun fun (cdr l))
      )
   )
)

(apply-fun (lambda (x) (* x 2)) '(1 2 3))   ; scale list
(apply-fun (lambda (x) (* x x)) '(1 2 3))   ; square list

for-each loop

(defun for-each(fun l)
   (if (null l) (print "done")
         (progn (funcall fun (car l))
                (for-each fun (cdr l))
         )

   )
)

(for-each (lambda (x) (print x)) '(1 2 3))

  (interactive)
  (insert "<p></p>")
  (backward-char 4)
)

  • Insert Befor/After Selection. (interactive "r") tells emacs to use the begin and the end of the current region as paramters. 

    (defun wrap-markup-region (start end)
  "Insert a markup <b></b> around a region."
  (interactive "r")
  (save-excursion
    (goto-char end) (insert "</b>")
    (goto-char start) (insert "<b>")
  )
)

Lambda Calculus

;-----------------------
(defun inc (x)
   (+ x 1)
)
;-----------------------
(defun zero (f x)
   x
)
;-----------------------
(defun one (f x)
   (funcall f x)
)
;-----------------------
(defun two (f x)
   (funcall f (funcall f x))
)
;-----------------------
;-----------------------
(zero 'inc 4)
(one 'inc 4)
(two 'inc 4)

(two (lambda (x) (* x x)) 4)
